Use this code to center text:

Add this code for centered text:

<p style="text-align: center;">text</p>

or this:

<div style="text-align: center;"></div>


Use this code for adding the font:

<head><link rel="preconnect" href="https://fonts.googleapis.com">
<link rel="preconnect" href="https://fonts.gstatic.com" crossorigin>
<link href="https://fonts.googleapis.com/css2?family=Comic+Relief:wght@400;700&display=swap" rel="stylesheet">
<style>body {
  font-family: 'Comic Relief', sans-serif;
}
</style></head>

Use this code for making links:

<a href="Link" style="text-decoration:none; font-size: 175%;">Link Text</a>


Use this code to make text bigger:

style="font-size: num%;"


Use this code for automatic dark mode:

<meta name="color-scheme" content="light dark" />


This adds an embed preview:

<head>
  <meta charset="utf-8">

  <!-- Open Graph (link preview) -->
  <meta property="og:title" content="My Website Title">
  <meta property="og:description" content="A short description of what this page is about.">
  <meta property="og:image" content="https://yourusername.github.io/path/preview.png">
  <meta property="og:url" content="https://yourusername.github.io/path/page.html">
  <meta property="og:type" content="website">

  <!-- Optional but recommended -->
  <meta name="theme-color" content="#ff0000">
</head>


This adds link Icons

document.querySelectorAll("a").forEach(link => {
        if (link.querySelector("img")) return;
        let domain;
        try {
          domain = new URL(link.href).origin;
        } catch {
          return;
        }
        const icon = document.createElement("img");
        icon.src = `https://www.google.com/s2/favicons?sz=32&domain_url=${domain}`;
        icon.alt = "favicon";
        icon.style.width = "1em"; //width of icon
        icon.style.height = "1em"; //height of icon
        icon.style.verticalAlign = "middle"; //where it aligns with text
        icon.style.marginTop = "-.4em"; //offset from the top of text
        icon.style.marginRight = "6px"; //offset from left edge of text
        icon.style.borderRadius = "25%"; //corner rounding
        link.prepend(icon);
});

or, for a section,

<div id="link-section">
  <a href="link">link</a>
</div>

and also add this below everything:

<script>
  document.querySelectorAll("#link-section a").forEach(link => {
    if (link.querySelector("img")) return;

    let domain;
    try {
      domain = new URL(link.href).origin;
    } catch {
      return;
    }

    const icon = document.createElement("img");
    icon.src = `https://www.google.com/s2/favicons?sz=32&domain_url=${domain}`;
    icon.alt = "favicon";
    icon.style.width = "1em";          // width of icon
    icon.style.height = "1em";         // height of icon
    icon.style.verticalAlign = "middle";
    icon.style.marginTop = "-.4em";
    icon.style.marginRight = "6px";
    icon.style.borderRadius = "25%";

    link.prepend(icon);
  });
</script>
